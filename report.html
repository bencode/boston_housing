def performance_metric2(y_true, y_predict):
    """"""
    mean = reduce(lambda s, x: s + x, y_true) / len(y_true)
    sstot = reduce(lambda acc, x: acc + (x - mean) ** 2, y_true, 0)
    ssres = reduce(lambda acc, (x, y): acc + (x - y) ** 2, zip(y_true, y_predict), 0)
    score = 1 - ssres / sstot

    return score
